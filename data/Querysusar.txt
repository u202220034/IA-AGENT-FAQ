#1 Creacion de Tablas (Ejecutar en el SQL, dentro de DataBase Explorer o Cloud Central)
-- Preguntas
CREATE COLUMN TABLE DBADMIN.CHATBOT_FAQ_QUESTIONS (
  QID              INTEGER,           
  AID              INTEGER PRIMARY KEY,
  QUESTION         NVARCHAR(4000),
  STATUS           NVARCHAR(50),      
  CREATED_AT       TIMESTAMP,         
  CREATED_BY       NVARCHAR(100),     
  QUESTION_VECTOR  REAL VECTOR(768)   
);

-- Respuestas
CREATE COLUMN TABLE DBADMIN.CHATBOT_FAQ_ANSWERS (
  AID      INTEGER PRIMARY KEY,
  ANSWER   NCLOB
);

-- Secuencias
CREATE SEQUENCE DBADMIN.CHATBOT_FAQ_AID_SEQ
  START WITH 1024
  INCREMENT BY 1;

#3 Usar el siguiente query para la creacion de los vectores de las preguntas.(Ejecutar en el SQL, dentro de DataBase Explorer o Cloud Central)

UPDATE DBADMIN.CHATBOT_FAQ_QUESTIONS
   SET QUESTION_VECTOR = VECTOR_EMBEDDING(QUESTION, 'QUERY', 'SAP_NEB.20240715')
 WHERE QUESTION_VECTOR IS NULL
   AND QUESTION IS NOT NULL;
